#include <bits/stdc++.h>
#define bit_capacity 100001

class bit{
    public:
        int length;
        // æ›´æ–°å…ƒç´ å€¼
        void update(int i,int delta);
        // å‰ç¶´å’Œ
        int query(int i);
        // å€é–“æŸ¥è©¢
        int query(int f,int l);
        // å¯¦éš›å„²å­˜çš„é™£åˆ—
        int *bitree=index;
    private:
        int lowbit(int x) {return x&(-x);}
        int index[bit_capacity];
};

void bit::update(int i,int delta){
    while(i<=length){
        index[i]+=delta;
        i+=lowbit(i);
    }
}

int bit::query(int i){
    int sum =0;
    while(i>0){
        sum+=index[i];
        i-=lowbit(i);
    }
    return sum;
}
int bit::query(int f,int l){
    return query(l)-query(f);
}

    tree.length=10;

        std::cout<<"BIT["<<i<<"]="<<tree.bitree[i]<<std::endl;
   }
    return 0;
}
